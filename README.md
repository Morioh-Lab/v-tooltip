### Install

```
npm install @morioh/v-tooltip
```

```js
import '@morioh/v-tooltip/dist/v-tooltip.css';

import Vue from 'vue'
import Tooltip from '@morioh/v-tooltip'

// global register
Vue.use(Tooltip);

```
